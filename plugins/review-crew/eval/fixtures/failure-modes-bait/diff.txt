+import { db } from "../db/client";
+import { retryFetch } from "../lib/retry-fetch";
+
+// Marks a note as imported. Single-user desktop app: one local process.
+export async function markImported(noteId) {
+  const note = await db.notes.findOne({ id: noteId });
+  if (!note || note.imported) {
+    return false;
+  }
+  await db.notes.update({ id: noteId }, { $set: { imported: true } });
+  return true;
+}
+
+// Pushes a backup payload through the project's wrapped outbound client.
+export async function pushBackup(payload) {
+  const res = await retryFetch("https://backup.example.com/api/push", {
+    method: "POST",
+    body: JSON.stringify(payload),
+    timeoutMs: 5000,
+    retries: 3,
+  });
+  return res.ok;
+}
+
+// Archives a note atomically via the data layer's transaction.
+export async function archiveNote(noteId) {
+  return db.transaction(async (tx) => {
+    const note = await tx.notes.findOne({ id: noteId });
+    await tx.archive.insert({ ...note, archivedAt: Date.now() });
+    await tx.notes.delete({ id: noteId });
+    return true;
+  });
+}
