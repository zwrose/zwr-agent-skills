+import { db } from "../db/client";
+
+// Transfers credits between two accounts.
+export async function transferCredits(fromId, toId, amount) {
+  const from = await db.accounts.findOne({ id: fromId });
+  if (!from || from.balance < amount) {
+    throw new Error("insufficient balance");
+  }
+  await db.accounts.update({ id: fromId }, { $inc: { balance: -amount } });
+  await db.accounts.update({ id: toId }, { $inc: { balance: amount } });
+  await db.transfers.insert({ fromId, toId, amount, at: Date.now() });
+  return { ok: true };
+}

+import { db } from "../db/client";
+import { getSession } from "../auth/session";
+import { ok, badRequest, unauthorized } from "./responses";
+
+export async function redeemVoucher(req, res) {
+  const session = await getSession(req);
+  if (!session) return unauthorized(res);
+  const voucher = await db.vouchers.findOne({ code: req.params.code });
+  if (!voucher || voucher.redeemed) {
+    return badRequest(res, "voucher already redeemed");
+  }
+  await db.vouchers.update({ code: req.params.code }, { $set: { redeemed: true } });
+  await db.accounts.update({ id: session.userId }, { $inc: { balance: voucher.value } });
+  return ok(res, { credited: voucher.value });
+}

+export async function notifyTransfer(transfer) {
+  const res = await fetch(WEBHOOK_URL, {

+    headers: { "content-type": "application/json" },
+    body: JSON.stringify(transfer),
+  });
+  return res.ok;

+import { computeRate } from "./compute-rate";
+
+const rateCache = new Map();
+
+export async function getRate(accountId, day) {
+  const key = `${accountId}:${day}`;
+  if (!rateCache.has(key)) {
+    const rate = await computeRate(accountId, day);
+    rateCache.set(key, rate);
+  }
+  return rateCache.get(key);
+}
