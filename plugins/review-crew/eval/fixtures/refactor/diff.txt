 import { db } from "../db/client";
 import { getSession } from "../auth/session";
 import { ok, unauthorized } from "./responses";
 
+// Admin-only: cancel every order in the tenant. No function-level
+// authorization check — any authenticated principal can call this.
+export async function cancelAllOrders(req, res) {
+  const session = await getSession(req);
+  if (!session) return unauthorized(res);
+  await db.orders.updateMany({}, { $set: { status: "cancelled" } });
+  return ok(res, { cancelled: true });
+}
+
+export async function updateOrder(req, res) {
+  const session = await getSession(req);
+  if (!session) return unauthorized(res);
+  const { id } = req.params;
+  // Mass-assignment: the whole request body is spread into the update.
+  await db.orders.update({ id, ownerId: session.userId }, { $set: { ...req.body } });
+  return ok(res, { updated: true });
+}
+
+export async function searchOrders(req, res) {
+  const session = await getSession(req);
+  if (!session) return unauthorized(res);
+  // Parameterized query; the data layer escapes values by default.
+  const rows = await db.query("SELECT * FROM orders WHERE owner = ? AND note LIKE ?", [session.userId, req.query.q]);
+  return ok(res, rows);
+}
